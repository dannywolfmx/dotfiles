*** Profile (.profile file)

#+begin_src sh
PROFILE_FILE=$HOME/.profile
if test -f "$PROFILE_FILE";
then
	echo -n "A .profile file detected, do you want to overwrite it? [Y/n] "
	read answer
	if [ "$answer" != "${answer#[Yy]}" ]
	then
		rm $HOME/.profile
		ln -s $PWD/.profile $PROFILE_FILE
	else
		echo "Skipped"
	fi
else
	#Link the profile
	ln -s $PWD/.profile $PROFILE_FILE
fi
#+end_src
